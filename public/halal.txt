<script>
	
$(document).ready(function () {
    readRecords(); 
	});

	function addRecord(){

		var courseId =  $("#courseId").val();
		var courseName =  $("#courseName").val();
		var coursePeriod =  $("#coursePeriod").val();

		$.ajax({

			url:"/courses/",
			type:'POST',
			data: {
				courseId:courseId,
				courseName:courseName,
				coursePeriod:coursePeriod,
			},
			success:function(data, status){
				readRecords();
			},

		});

	}

//////////////////Display Records
	function readRecords(){
		
		var readrecords = "readrecords";
		$.ajax({
			url:"/courses",
			type:"POST",
			data:{readrecords:readrecords},
			success:function(data,status){
				$('#records_content').html(data);
			},

		});
	}


/////////////delete userdetails ////////////
function DeleteUser(deleteid){

	var conf = confirm("are u sure");
	if(conf == true) {
	$.ajax({
		url:"/courses/:id",
		type:'POST',
		data: {  deleteid : deleteid},

		success:function(data, status){
			readRecords();
		}
	});
	}
}



function GetUserDetails(id){
	  $("#hidden_user_id").val(id);
	  $.post("/courses/", {
            id: id
        },
        function (data, status) {
            alert(data);
            //JSON.parse() parses a string, written in JSON format, and returns a JavaScript object.
            var user = JSON.parse(data);
            alert(user);

            $("#update_courseId").val(user.courseId);
            $("#update_courseName").val(user.courseName);
            $("#update_coursePeriod").val(user.coursePeriod);
        }
    );
    $("#update_user_modal").modal("show");
}




function UpdateUserDetails() {
    var courseId = $("#update_courseId").val();
    var courseName = $("#update_courseName").val();
    var coursePeriod = $("#update_coursePeriod").val();
    var hidden_user_id = $("#hidden_user_id").val();
    $.post("/courses/", {
            hidden_user_id: hidden_user_id,
            courseId:courseId,
			courseName:courseName,
			coursePeriod:coursePeriod
        },
        function (data, status) {
            $("#update_user_modal").modal("hide");
            readRecords();
        }
    );
}
